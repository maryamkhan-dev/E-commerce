CREATE FUNCTION UsernameExists(@username VARCHAR(50)) RETURNS BIT
AS
BEGIN
    DECLARE @user_count INT;
    SELECT @user_count = COUNT(*) FROM Users WHERE Username = @username;
    IF @user_count > 0
        RETURN 1;
    ELSE
        RETURN 0;
    
    -- default return statement
    RETURN 0;
END;

-- function
CREATE FUNCTION CheckUser(@username VARCHAR(50), @password VARCHAR(50), @type VARCHAR(50)) RETURNS BIT
AS
BEGIN
    DECLARE @user_count INT;

    SELECT @user_count = COUNT(*) FROM Users 
    WHERE Username = @username AND Password = @password AND User_Type = @type;

    IF @user_count > 0
        RETURN 1;
    ELSE
        RETURN 0;

	-- default return statement
    RETURN 0;
END;

CREATE PROCEDURE InsertCartItem @CustomerID VARCHAR(50), @BookISBN VARCHAR(50)
AS
BEGIN
    -- Insert item into cart
    INSERT INTO CartItems (CartID, BookISBN, Quantity)
    VALUES ((SELECT CartID FROM Cart WHERE CustomerID = @CustomerID), @BookISBN, 1);
END;


-- three table joing (3tj)
CREATE PROCEDURE GetCartItems @CustomerID VARCHAR(50)
AS
BEGIN
    -- Retrieve cart items for the given customer
    SELECT ci.CartID, ci.BookISBN, ci.Quantity, b.Title, b.Author, b.Price
    FROM CartItems ci
    JOIN Cart c ON ci.CartID = c.CartID
    JOIN Books b ON ci.BookISBN = b.ISBN
    WHERE c.CustomerID = @CustomerID;
END;



CREATE PROCEDURE GetCustomerOrders @CustomerID VARCHAR(50)
AS
BEGIN
    SELECT OrderID, CustomerID, Order_Date, Status, Order_Total, Delivery_fee, couponCode
    FROM Orders
    WHERE CustomerID = @CustomerID
    ORDER BY Order_Date DESC;
END;


-- three table join (3tj)
CREATE PROCEDURE GetOrderItems @OrderID INT
AS
BEGIN
    SELECT oi.Quantity, oi.Title, oi.Price, oi.BookISBN,
           CASE WHEN r.OrderID IS NOT NULL THEN 'Returned' ELSE 'Delivered' END AS Status
    FROM OrderItemsView oi
    LEFT JOIN Returns r ON oi.OrderID = r.OrderID AND oi.BookISBN = r.BookISBN
    WHERE oi.OrderID = @OrderID;
END;

-- two table join (2tj)
CREATE PROCEDURE DeleteCartItem @CustomerID VARCHAR(50), @BookISBN VARCHAR(20)
AS
BEGIN
    DELETE ci
    FROM CartItems ci
    INNER JOIN Cart c ON ci.CartID = c.CartID
    WHERE c.CustomerID = @CustomerID AND ci.BookISBN = @BookISBN;
END;



-- two table join (2tj)
CREATE PROCEDURE GetCustomerDetails @Username VARCHAR(50)

-- view

CREATE VIEW OrderItemsView
AS
SELECT oi.OrderID, oi.BookISBN, oi.Quantity, b.Title, b.Author, b.Price
FROM OrderItems oi
JOIN Books b ON oi.BookISBN = b.ISBN;

-- two table join (2tj)  used twice
-- aggregate (ag)
-- nested subquery (nsq)
CREATE FUNCTION CalculateOrderTotal (@OrderID INT)
RETURNS DECIMAL(10, 2)
AS
BEGIN
    DECLARE @Total DECIMAL(10, 2);

    SELECT @Total = SUM(oi.Quantity * b.Discounted_Price)
    FROM OrderItems oi
    INNER JOIN Books b ON oi.BookISBN = b.ISBN
    WHERE oi.OrderID = @OrderID;

    -- for coupon discount amount
    SELECT @Total = @Total - ISNULL((
        SELECT c.Discount_Amount
        FROM Coupons c
        INNER JOIN Orders o ON c.CouponCode = o.couponCode
        WHERE o.OrderID = @OrderID
    ), 0);

    -- delivery fee
    SELECT @Total = @Total + ISNULL((
        SELECT Delivery_fee
        FROM Orders
        WHERE OrderID = @OrderID
    ), 0);

    RETURN @Total;
END;


-- group by (group)
CREATE PROCEDURE GetAverageOrderValuePerCustomer
AS
BEGIN
    SELECT o.CustomerID, AVG(o.Order_Total) AS AverageOrderValue
    FROM Orders o
    GROUP BY o.CustomerID;
END;


-- aggregate (ag)
CREATE PROCEDURE GetAverageNumberOfItemsPerOrder
AS
BEGIN
    SELECT AVG(Quantity) AS AverageNumberOfItems
    FROM OrderItems;
END;


-- two table join (2tj)
-- group by (group)
CREATE PROCEDURE GetRevenueByEachCustomer
AS
BEGIN
    SELECT c.Username, SUM(o.Order_Total) AS Revenue
    FROM Orders o
    JOIN Customer c ON o.CustomerID = c.Username
    GROUP BY c.Username;
END;


-- group by (group)
-- two table join (2tj)
-- aggregate (ag)
CREATE PROCEDURE GetMostBoughtBook
AS
BEGIN
    WITH MaxPurchaseCount AS (
        SELECT TOP 1 BookISBN, COUNT(*) AS PurchaseCount
        FROM OrderItems
        GROUP BY BookISBN
        ORDER BY COUNT(*) DESC
    )
    SELECT mp.BookISBN, b.Title, mp.PurchaseCount
    FROM MaxPurchaseCount mp
    JOIN Books b ON mp.BookISBN = b.ISBN;
END;


-- aggregate (ag)

-- aggregate (ag)
-- three table join (3tj)
-- group by (group)
CREATE PROCEDURE GetRevenueByEachBookNotReturned
AS
BEGIN
    SELECT b.Title AS BookTitle, b.ISBN AS BookISBN, SUM(oi.Quantity * b.Price) AS Revenue
    FROM OrderItems oi
    JOIN Books b ON oi.BookISBN = b.ISBN
    LEFT JOIN Returns r ON oi.OrderID = r.OrderID AND oi.BookISBN = r.BookISBN
    WHERE r.OrderID IS NULL
    GROUP BY b.Title, b.ISBN;
END;


-- three table join (3tj)
CREATE PROCEDURE ClearCartItems @CustomerID VARCHAR(50)
AS
BEGIN
    DELETE FROM CartItems
	FROM CartItems
	INNER JOIN Carts ON CartItems.cartid = Carts.id
	INNER JOIN Customers ON Carts.customerid = Customers.id
	WHERE Customers.username = @CustomerID;
END;
